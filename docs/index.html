2) Spooky Smoke Animation (spooky-smoke-animation.tsx)

WebGL2 fragment shader-based animated background. You can tint the smoke with a color prop.

import { SmokeBackground } from "@/components/ui/spooky-smoke-animation";

function Demo() {

  return <SmokeBackground smokeColor="#FF00AA" />;
}
